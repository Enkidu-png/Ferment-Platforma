    **src/app/auth/confirm/route.ts — CREATE NEW FILE:**

    This is the PKCE email confirmation callback. Supabase sends users to `<siteURL>/auth/confirm?token_hash=...&type=email` after they click the verification link in their email.

    ```typescript
    import { createClient } from '@/lib/supabase/server';
    import { type NextRequest, NextResponse } from 'next/server';

    export async function GET(request: NextRequest) {
      const { searchParams, origin } = new URL(request.url);
      const token_hash = searchParams.get('token_hash');
      const type = searchParams.get('type') as 'email' | null;

      if (token_hash && type) {
        const supabase = await createClient();
        const { data, error } = await supabase.auth.verifyOtp({
          type,
          token_hash,
        });

        if (!error && data.user) {
          // Extract shop_name from user metadata (set during signUp)
          const shopName = data.user.user_metadata?.shop_name as string | undefined;

          // Generate a slug from shop_name — same transform as registerSchema
          const baseSlug = shopName?.toLowerCase().replace(/\s+/g, '-') ?? data.user.id;

          // Check for slug conflicts and append suffix if needed
          const { data: existing } = await supabase
            .from('tenants')
            .select('slug')
            .eq('slug', baseSlug)
            .maybeSingle();

          const slug = existing
            ? `${baseSlug}-${Math.random().toString(36).slice(2, 6)}`
            : baseSlug;

          // Create pending tenant row — only after email is confirmed
          // stripe_account_id omitted (nullable — confirmed in types.ts before implementing)
          await supabase.from('tenants').insert({
            name: shopName ?? 'New Store',
            slug,
            status: 'pending',
            user_id: data.user.id,
          });

          return NextResponse.redirect(`${origin}/pending`);
        }
      }

      // Token invalid or expired — redirect back to sign-in with error
      return NextResponse.redirect(`${origin}/sign-in?error=confirmation_failed`);
    }
    ```

    IMPORTANT: Before writing the `insert` call, check `src/lib/supabase/types.ts` for the exact column names and nullable constraints on the `tenants` table. If `stripe_account_id` is NOT NULL without a default, omit the insert line and add a TODO comment — do not let the handler fail silently.
