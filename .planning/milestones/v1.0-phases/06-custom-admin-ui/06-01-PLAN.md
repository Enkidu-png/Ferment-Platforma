**Step A: Add `adminProcedure` to src/trpc/init.ts**

After the existing `protectedProcedure`, add:

```typescript
export const adminProcedure = t.procedure.use(async ({ ctx, next }) => {
  if (!ctx.user) {
    throw new TRPCError({ code: 'UNAUTHORIZED', message: 'Not authenticated' });
  }
  // app_role is embedded by custom_access_token_hook into user.app_metadata
  // The hook's coalesce fix (Phase 3) ensures app_role is present for seeded admin
  const appRole = ctx.user.app_metadata?.app_role;
  if (appRole !== 'super-admin') {
    throw new TRPCError({ code: 'FORBIDDEN', message: 'Super-admin access required' });
  }
  return next({ ctx: { ...ctx, user: ctx.user } });
});
```

Important: `supabaseAdmin` is NOT needed here — we are reading from the JWT claim (`app_metadata`), not making a DB query. The `custom_access_token_hook` already embedded `app_role` in Phase 1/3.

```typescript
'use client';
import Link from 'next/link';
import { usePathname } from 'next/navigation';

const links = [
  { label: 'Merchants', href: '/admin/merchants' },
  { label: 'Products', href: '/admin/products' },
  { label: 'Categories', href: '/admin/categories' },
  { label: 'Tags', href: '/admin/tags' },
  { label: 'Orders', href: '/admin/orders' },
];

export function AdminSidebarNav() {
  const pathname = usePathname();
  return (
    <nav className="w-48 shrink-0 border-r h-screen p-4 flex flex-col gap-1">
      <p className="text-xs font-semibold uppercase text-muted-foreground mb-2">Admin</p>
      {links.map(({ label, href }) => (
        <Link
          key={href}
          href={href}
          className={`block px-3 py-2 rounded-md text-sm hover:bg-accent ${
            pathname.startsWith(href) ? 'bg-accent font-semibold' : ''
          }`}
        >
          {label}
        </Link>
      ))}
    </nav>
  );
}
```

**Step C: Create admin layout**

Create `src/app/(admin)/admin/layout.tsx` — a Server Component that:
1. Creates Supabase client via `await createClient()`
2. Calls `getUser()` — if no user, `redirect('/sign-in')`
3. Reads `user.app_metadata?.app_role` — if not `'super-admin'`, `redirect('/')`
4. Renders `<AdminSidebarNav />` + `{children}` in a flex layout

```typescript
import { redirect } from 'next/navigation';
import { createClient } from '@/lib/supabase/server';
import { AdminSidebarNav } from '@/modules/admin/ui/components/admin-sidebar-nav';

export default async function AdminLayout({ children }: { children: React.ReactNode }) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();

  if (!user) {
    redirect('/sign-in');
  }

  const appRole = user.app_metadata?.app_role;
  if (appRole !== 'super-admin') {
    redirect('/');
  }

  return (
    <div className="flex min-h-screen">
      <AdminSidebarNav />
      <main className="flex-1 overflow-y-auto p-6">
        {children}
      </main>
    </div>
  );
}
```
