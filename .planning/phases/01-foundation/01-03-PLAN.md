  <files></files>
  <action>
    Run the hook function SQL and permission grants in the SQL Editor, then register the hook in the Authentication dashboard.

    STEP A — Run the hook SQL (SQL Editor > New query):

    Paste and run this SQL block:

    -- JWT CUSTOM CLAIMS HOOK FUNCTION
    -- Source: https://supabase.com/docs/guides/auth/auth-hooks/custom-access-token-hook
    -- Runs every time a JWT is minted (login, token refresh).
    -- Embeds tenant_id and app_role into the token so RLS policies can read them
    -- without querying the database on every request.
    create or replace function public.custom_access_token_hook(event jsonb)
    returns jsonb
    language plpgsql
    stable
    as $$
    declare
      claims        jsonb;
      user_tenant_id uuid;
      user_role     text;
    begin
      -- Fetch tenant membership (null for buyers who have no tenant row)
      select ut.tenant_id into user_tenant_id
      from public.user_tenants ut
      where ut.user_id = (event->>'user_id')::uuid
      limit 1;

      -- Fetch app role from users table
      select u.role into user_role
      from public.users u
      where u.id = (event->>'user_id')::uuid;

      -- Embed claims into the JWT payload
      claims := event->'claims';
      claims := jsonb_set(claims, '{tenant_id}', to_jsonb(user_tenant_id));
      claims := jsonb_set(claims, '{app_role}',  to_jsonb(coalesce(user_role, 'user')));

      return jsonb_set(event, '{claims}', claims);
    end;
    $$;

    -- GRANT REQUIRED PERMISSIONS
    -- supabase_auth_admin must be able to execute the hook and read the referenced tables
    grant usage on schema public to supabase_auth_admin;
    grant execute on function public.custom_access_token_hook to supabase_auth_admin;
    grant all on table public.user_tenants to supabase_auth_admin;
    grant all on table public.users to supabase_auth_admin;

    -- Lock the function from other roles (anon/authenticated cannot call it directly)
    revoke execute on function public.custom_access_token_hook from authenticated, anon, public;

    STEP B — Register the hook in the Supabase dashboard:

    1. In the left sidebar, go to: Authentication > Hooks
    2. Find "Custom Access Token" hook type.
    3. Click "Enable hook" or the toggle next to it.
    4. In the "Hook type" dropdown, select "Postgres function".
    5. In the function dropdown, choose: public.custom_access_token_hook
    6. Click "Save".
    7. Confirm the hook shows a green checkmark or "Active" status.
  </action>
