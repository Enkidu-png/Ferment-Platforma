```typescript
config({ path: '.env', override: false })

export default defineConfig({

  timeout: 30_000,
  retries: 0,
  fullyParallel: false, // run tests sequentially to avoid port conflicts
  use: {

    command: 'npm run dev',
    url: 'http://localhost:3000',
    reuseExistingServer: !process.env.CI, // reuse a running dev server locally; always start fresh in CI
    timeout: 120_000,
  },
```

```typescript
// tests/smoke/auth.spec.ts
// Auth route smoke tests — verifies the Phase 2 rewired sign-in/sign-up/confirm/pending pages
import { test, expect } from '@playwright/test'

// Each test starts with a fresh browser context (no shared session state)

test('sign-in page renders with email and password fields', async ({ page }) => {
  await page.goto('/sign-in')
  await expect(page.getByLabel(/email/i)).toBeVisible()
  await expect(page.getByLabel(/password/i)).toBeVisible()
  await expect(page.getByRole('button', { name: /sign in/i })).toBeVisible()
})

test('admin can sign in and is redirected away from /sign-in', async ({ page }) => {
  await page.goto('/sign-in')
  await page.getByLabel(/email/i).fill(process.env.SEED_ADMIN_EMAIL!)
  await page.getByLabel(/password/i).fill(process.env.SEED_ADMIN_PASSWORD!)
  await page.getByRole('button', { name: /sign in/i }).click()
  // After successful login, should not stay on /sign-in
  await page.waitForURL(url => !url.pathname.includes('/sign-in'), { timeout: 10_000 })
  await expect(page).not.toHaveURL(/\/sign-in/)
})

test('sign-up page renders with shopName field', async ({ page }) => {
  await page.goto('/sign-up')
  // Phase 2 renamed 'username' to 'shopName' in the form
  await expect(page.getByLabel(/shop name/i)).toBeVisible()
  await expect(page.getByLabel(/email/i)).toBeVisible()
  await expect(page.getByLabel(/password/i)).toBeVisible()
})

test('/pending page renders without 404', async ({ page }) => {
  const response = await page.goto('/pending')
  expect(response?.status()).not.toBe(404)
  // The page should contain some visible content
  await expect(page.locator('body')).not.toBeEmpty()
})

test('/auth/confirm route responds without 404', async ({ page }) => {
  // Navigate with no token — will redirect or show error, but must not 404
  const response = await page.goto('/auth/confirm')
  expect(response?.status()).not.toBe(404)
})
```

```typescript
// tests/smoke/storefront.spec.ts
// Storefront smoke tests — verifies subdomain routing, product listing, and category filter
// Depends on seed data from Plan 01: ceramics-by-ana tenant with pottery products
import { test, expect } from '@playwright/test'

test('ceramics-by-ana storefront loads via subdomain', async ({ page }) => {
  // Chromium resolves *.localhost natively — no /etc/hosts change needed
  await page.goto('http://ceramics-by-ana.localhost:3000')
  // Middleware should route this to the ceramics-by-ana storefront
  // The page should not be a 404 or error page
  await expect(page.locator('body')).not.toBeEmpty()
  // Should show the tenant name somewhere on the page
  await expect(page.getByText(/ceramics by ana/i)).toBeVisible({ timeout: 15_000 })
})

test('ceramics-by-ana storefront shows pottery products', async ({ page }) => {
  await page.goto('http://ceramics-by-ana.localhost:3000')
  // At least one product seeded for this tenant should be visible
  // Products seeded: Handmade Stoneware Mug, Ceramic Salad Bowl, etc.
  await expect(
    page.getByText(/handmade stoneware mug|ceramic salad bowl|espresso cup|ramen bowl/i)
  ).toBeVisible({ timeout: 15_000 })
})

test('ceramics-by-ana storefront does not show woodworks-jan products', async ({ page }) => {
  await page.goto('http://ceramics-by-ana.localhost:3000')
  await page.waitForLoadState('networkidle')
  // RLS should prevent cross-tenant product leakage
  await expect(page.getByText(/oak serving board|walnut key tray/i)).not.toBeVisible()
})

test('category filter narrows products on ceramics-by-ana storefront', async ({ page }) => {
  await page.goto('http://ceramics-by-ana.localhost:3000')
  await page.waitForLoadState('networkidle')
  // Click the pottery category filter (top-level category)
  const potteryFilter = page.getByRole('link', { name: /pottery/i }).first()
  if (await potteryFilter.isVisible()) {
    await potteryFilter.click()
    await page.waitForLoadState('networkidle')
    // After filtering, pottery products should still be visible
    await expect(
      page.getByText(/handmade stoneware mug|ceramic salad bowl|ramen bowl/i)
    ).toBeVisible({ timeout: 10_000 })
  } else {
    // If category filter uses a different mechanism, just verify products render
    await expect(page.getByText(/handmade stoneware mug/i)).toBeVisible()
  }
})
```
