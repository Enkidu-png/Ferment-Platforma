Create `scripts/send-password-resets.ts`. Use the inline service-role client pattern (same as `scripts/seed.ts`) — do NOT import `src/lib/supabase/admin.ts` because it uses `server-only` which is incompatible with Node.js script context.

Script requirements:
- Accept `--dry-run` flag: when set, log each artist email without sending any email
- Filter out super-admin users (check `u.app_metadata?.app_role === 'super-admin'`)
- Send via `supabase.auth.resetPasswordForEmail()` in live mode
- Add 1500ms delay between sends (Supabase Free tier: 4 emails/hour limit)
- Log clearly: `[DRY RUN] Would send reset to: email@x.com` or `SENT email@x.com` or `FAIL email@x.com: error`
- Exit with code 1 if `listUsers()` fails; exit 0 on success (even if some individual sends fail — log failures)

```typescript
// scripts/send-password-resets.ts
// Run: npx tsx --env-file=.env.local --env-file=.env scripts/send-password-resets.ts [--dry-run]
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!,
  { auth: { autoRefreshToken: false, persistSession: false } }
)

const dryRun = process.argv.includes('--dry-run')

async function sendPasswordResets() {
  const { data: { users }, error } = await supabase.auth.admin.listUsers({
    page: 1,
    perPage: 1000,
  })
  if (error) throw error

  const artists = users.filter(u =>
    u.app_metadata?.app_role !== 'super-admin' && u.email
  )

  console.log(`Found ${artists.length} artists. Dry run: ${dryRun}`)

  for (const user of artists) {
    if (!user.email) continue
    if (dryRun) {
      console.log(`[DRY RUN] Would send reset to: ${user.email}`)
      continue
    }
    const { error: resetError } = await supabase.auth.resetPasswordForEmail(user.email)
    if (resetError) {
      console.error(`FAIL ${user.email}: ${resetError.message}`)
    } else {
      console.log(`SENT ${user.email}`)
    }
    // Respect Supabase Free tier 4 emails/hour limit
    await new Promise(r => setTimeout(r, 1500))
  }

  console.log('Done.')
}

sendPasswordResets().catch(err => { console.error(err); process.exit(1) })
```
